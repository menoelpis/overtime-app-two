```erb
    <div class="col-sm-3">

          <div class='input-group date' id='datetimepicker1'>
              <%= f.date_field :date %>

                  </span>
              </span>

    </div>
    <div class="col-sm-3">

            <div class='input-group text'>

	<%= f.submit 'Save' %>

<% end %>

<script type="text/javascript">
    $(function () {
        $('#datetimepicker1').datetimepicker({
            viewMode: 'years',
            format: 'DD/MM/YYYY'
        });
    });
</script>
```

```rb
end
.
.
```
